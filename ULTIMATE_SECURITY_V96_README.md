## 🔥 USAGE EXAMPLES

### Basic Integration
```python
from ava.security.blockchain_security import BlockchainSecurityLedger
from ava.security.tangle_network import TangleSecurityNetwork
from ava.security.iot_security_enhanced import ZeroTrustIoTAuthenticator

# Initialize
blockchain = BlockchainSecurityLedger(difficulty=4)
tangle = TangleSecurityNetwork()
auth = ZeroTrustIoTAuthenticator()

# Register device
device_id = "camera_001"
credentials = auth.issue_credentials(device_id, ["read", "write"])

# Report threat
tangle.add_threat_transaction({
    'type': 'INTRUSION',
    'severity': 'HIGH',
    'device': device_id
})

# Mine to blockchain
blockchain.add_threat_to_ledger({'type': 'CONFIRMED_THREAT'})
```

### Complete Platform
```python
from launch_ultimate_v96 import UltimateSecurityPlatformV96

# Initialize complete platform
platform = UltimateSecurityPlatformV96()

# Register secure IoT device
device = platform.register_secure_iot_device(
    device_type="camera",
    ip_address="192.168.1.100",
    firmware_data=b"FIRMWARE_v1.0.0"
)

# Process threats (full chain: consensus → tangle → blockchain → response)
platform.process_threat({
    'type': 'MALWARE',
    'severity': 'CRITICAL'
}, source_device_id=device['device_id'])

# Get statistics
stats = platform.get_comprehensive_statistics()
print(f"Threats Blocked: {stats['platform']['total_threats_blocked']}")
```
